** Imperative programming
*** Imperative factorial
#+BEGIN_SRC c
int n = 23;
int f = 1;
int i = 1;

iter:
f = f * i;
i++;
if (i <= n)
  goto iter;

print f;
#+END_SRC

*** More structure
#+BEGIN_SRC c
int n = 23;
int f = 1;
int i = 1;

iter:
if (i <= n) {
  f = f * i;
  i++;
  goto iter;
}

print f;
#+END_SRC

*** Move flow control
#+BEGIN_SRC c
int n = 23;
int f = 1;
int i;

for (i = 1; i <= n;) {
  f = f * i;
  i++;
}

print f;
#+END_SRC

*** Procedural
#+BEGIN_SRC c
int f;

int factorial(int n) {
  int i;
  for (i = 1; i <= n; i++) {
    f = f * i;
  }
}

int main() {
  f = 1;
  factorial(23);
  printf("%d", f)
}
#+END_SRC
