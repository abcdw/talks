Imperative factorial

  goto iter;

More structure

int n = 23;
int f = 1;
int i = 1;

iter:
if (i <= n) {
  f = f * i;
  i++;
  goto iter;
}

print f;

Move flow control

int n = 23;
int f = 1;
int i;

for (i = 1; i <= n;) {
  f = f * i;
  i++;
}

print f;

Procedural

int f;

int factorial(int n) {
  int i;
  for (i = 1; i <= n; i++) {
    f = f * i;
  }
}

int main() {
  f = 1;
  factorial(23);
  printf("%d", f)
}
